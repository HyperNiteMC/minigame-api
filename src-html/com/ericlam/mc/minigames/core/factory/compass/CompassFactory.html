001package com.ericlam.mc.minigames.core.factory.compass;
002
003import com.ericlam.mc.minigames.core.factory.Factory;
004import com.ericlam.mc.minigames.core.game.GameTeam;
005
006/**
007 * 羅盤追蹤器工廠
008 * <p>
009 * 如果沒有隊伍，則會追蹤最近的一位玩家
010 * </p>
011 * <p>
012 * 否則，追蹤目標隊伍最近的一位玩家
013 * </p>
014 */
015public interface CompassFactory extends Factory<CompassTracker> {
016
017    /**
018     * 設置隊伍追蹤目標
019     *
020     * @param team   隊伍
021     * @param target 目標隊伍
022     * @return this
023     */
024    CompassFactory setTeamTarget(GameTeam team, GameTeam target);
025
026    /**
027     * 設置追蹤範圍
028     *
029     * @param range 範圍
030     * @return this
031     */
032    CompassFactory setTrackerRange(int range);
033
034    /**
035     * 設置搜索文字標題動畫
036     *
037     * @param text 搜索文字標題動畫
038     * @return this
039     */
040    CompassFactory setSearchingText(String... text);
041
042    /**
043     * 設置捕獲到目標時的指南針標題
044     * <p>
045     * 佔位符如下:
046     * <ul>
047     *     <li> &lt;target&gt; - targetName </li>
048     *     <li> &lt;distance&gt; - distance between target </li>
049     *     <li> &lt;team&gt; - teamName </li>
050     * </ul>
051     *
052     * @param caughtText 捕獲到目標時的指南針標題
053     * @return this;
054     */
055    CompassFactory setCaughtText(String caughtText);
056
057}
