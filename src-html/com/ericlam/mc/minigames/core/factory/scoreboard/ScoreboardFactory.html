001package com.ericlam.mc.minigames.core.factory.scoreboard;
002
003import com.ericlam.mc.minigames.core.factory.Factory;
004import com.ericlam.mc.minigames.core.game.GameTeam;
005import org.bukkit.scoreboard.Team;
006
007/**
008 * 計分版工廠
009 */
010public interface ScoreboardFactory extends Factory<GameBoard> {
011
012    /**
013     * 添加隊伍設定
014     *
015     * @param gameTeam 隊伍
016     * @param option   設定
017     * @param status   狀態
018     * @return this
019     */
020    ScoreboardFactory addTeamSetting(GameTeam gameTeam, Team.Option option, Team.OptionStatus status);
021
022    /**
023     * 添加全局設定
024     * <p>
025     * 注意，此設定僅限能套用在<b>沒有隊伍</b>的玩家。
026     *
027     * @param option 設定
028     * @param status 狀態
029     * @return this
030     */
031    ScoreboardFactory addSetting(Team.Option option, Team.OptionStatus status);
032
033    /**
034     * 設置計分版標題, 支援顏色代碼
035     *
036     * @param title 標題
037     * @return this
038     */
039    ScoreboardFactory setTitle(String title);
040
041    /**
042     * 設置計分版內容
043     *
044     * @param key   此行內容的標識文字
045     * @param text  文字, 支援顏色
046     * @param score 分數
047     * @return this
048     */
049    ScoreboardFactory setLine(String key, String text, int score);
050
051    /**
052     * 添加計分版內容而不標識文字
053     *
054     * @param text  文字, 支援顏色
055     * @param score 分數
056     * @return this
057     */
058    ScoreboardFactory addLine(String text, int score);
059
060
061}
