001package com.ericlam.mc.minigames.core.manager;
002
003import com.ericlam.mc.minigames.core.arena.Arena;
004import com.ericlam.mc.minigames.core.arena.CreateArena;
005import com.ericlam.mc.minigames.core.exception.NoMoreElementException;
006import com.ericlam.mc.minigames.core.exception.arena.create.*;
007import com.google.common.collect.ImmutableList;
008import org.bukkit.Location;
009import org.bukkit.World;
010import org.bukkit.entity.Player;
011
012import java.util.List;
013import java.util.concurrent.CompletableFuture;
014
015/**
016 * 場地設置管理器
017 */
018public interface ArenaCreateManager {
019
020    /**
021     * 設置作者
022     *
023     * @param arena  場地
024     * @param author 作者
025     * @throws ArenaNotExistException 場地不存在時
026     */
027    void setAuthor(String arena, String author) throws ArenaNotExistException;
028
029    /**
030     * 設置顯示名稱
031     * @param arena 場地
032     * @param displayName 顯示名稱
033     * @throws ArenaNotExistException 場地不存在時
034     */
035    void setDisplayName(String arena, String displayName) throws ArenaNotExistException;
036
037    /**
038     * 設置場地名稱
039     * @param arena 場地
040     * @param newName 名稱
041     * @throws ArenaNotExistException 場地不存在時
042     * @throws ArenaNameExistException 場地名與其他場地重複時
043     */
044    void setName(String arena, String newName) throws ArenaNotExistException, ArenaNameExistException;
045
046    /**
047     * 添加位置
048     * @param arena 場地
049     * @param warp 地標
050     * @param location 新位置
051     * @throws ArenaNotExistException 場地不存在時
052     * @throws WarpNotExistException 地標不存在時
053     * @throws LocationMaxReachedException 最大設置數達到時
054     */
055    void addSpawn(String arena, String warp, Location location) throws ArenaNotExistException, WarpNotExistException, LocationMaxReachedException;
056
057    /**
058     * 刪除上一個位置
059     * @param arena 場地
060     * @param warp 地標
061     * @throws ArenaNotExistException 場地不存在時
062     * @throws WarpNotExistException 地標不存在時
063     * @throws NoMoreLocationException 沒有更多位置時
064     */
065    void removeSpawn(String arena, String warp) throws ArenaNotExistException, WarpNotExistException, NoMoreLocationException;
066
067    /**
068     * 新增地標
069     * @param arena 場地
070     * @param warp 新地標
071     * @throws ArenaNotExistException 場地不存在時
072     * @throws WarpExistException 地標已存在時
073     * @throws IllegalWarpException 地標不符合規範時
074     */
075    void createWarp(String arena, String warp) throws ArenaNotExistException, WarpExistException, IllegalWarpException;
076
077    /**
078     * 刪除地標
079     * @param arena 場地
080     * @param warp 地標
081     * @throws ArenaNotExistException 場地不存在時
082     * @throws WarpNotExistException 地標不存在時
083     */
084    void removeWarp(String arena, String warp) throws ArenaNotExistException, WarpNotExistException;
085
086    /**
087     * 保存場地
088     * @param arena 場地
089     * @return 異步加載
090     * @throws SetUpNotFinishException 設置尚未完成時
091     * @throws ArenaNotExistException 場地不存在時
092     * @throws ArenaUnchangedExcpetion 場地沒有被變更時
093     */
094    CompletableFuture<Boolean> saveArena(String arena) throws SetUpNotFinishException, ArenaNotExistException, ArenaUnchangedExcpetion;
095
096    /**
097     * 獲取場地資訊
098     * @param arena 場地
099     * @return 資訊列
100     * @throws ArenaNotExistException 場地不存在時
101     */
102    String[] getArenaInfo(String arena) throws ArenaNotExistException;
103
104    /**
105     * 刪除場地
106     * @param arena 場地
107     * @return 異步運作
108     * @throws ArenaNotExistException 場地不存在時
109     */
110    CompletableFuture<Boolean> deleteArena(String arena) throws ArenaNotExistException;
111
112    /**
113     * 創建場地
114     * @param name 場地名稱
115     * @param player 玩家
116     * @throws ArenaExistException 場地名稱已存在時
117     */
118    void createArena(String name, Player player) throws ArenaExistException;
119
120    /**
121     * 獲取該世界內存在的場地列
122     * @param world 世界
123     * @return 該世界內存在的場地列
124     */
125    Arena[] getArenasFromWorld(World world);
126
127    /**
128     * 獲取所有場地列表
129     * @return 無法被變更的場地列表
130     */
131    ImmutableList<Arena> getArenaList();
132
133    /**
134     * 獲取地標列表
135     * @param arena 場地
136     * @return 地標列表
137     * @throws ArenaNotExistException 場地不存在時
138     */
139    List<String> getWarpList(String arena) throws ArenaNotExistException;
140
141    /**
142     * 獲取該場地的編輯容器
143     * @param arena 場地
144     * @return 可編輯的場地容器
145     * @throws ArenaNotExistException 場地不存在時
146     */
147    CreateArena getCreateArena(String arena) throws ArenaNotExistException;
148
149    /**
150     * 新增場地描述
151     * @param arena 場地
152     * @param text 文字內容
153     * @throws ArenaNotExistException 場地不存在時
154     */
155    void addDescriptionLine(String arena, String text) throws ArenaNotExistException;
156
157    /**
158     * 刪除上一個場地描述
159     * @param arena 場地
160     * @throws NoMoreElementException 沒有更多描述時
161     * @throws ArenaNotExistException 場地不存在時
162     */
163    void removeDescriptionLine(String arena) throws NoMoreElementException, ArenaNotExistException;
164}
