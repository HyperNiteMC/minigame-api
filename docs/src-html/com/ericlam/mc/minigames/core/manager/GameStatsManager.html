001package com.ericlam.mc.minigames.core.manager;
002
003import com.ericlam.mc.minigames.core.character.GamePlayer;
004import com.ericlam.mc.minigames.core.exception.gamestats.PlayerNotExistException;
005import com.ericlam.mc.minigames.core.gamestats.GameStatsEditor;
006import org.bukkit.OfflinePlayer;
007import org.bukkit.entity.Player;
008
009import java.util.concurrent.CompletableFuture;
010
011/**
012 * 遊戲玩家資料管理器
013 */
014public interface GameStatsManager {
015
016    /**
017     * 獲取遊戲玩家資訊
018     *
019     * @param player 遊戲玩家
020     * @return 遊戲玩家資訊
021     * @throws PlayerNotExistException 遊戲玩家不存在時
022     */
023    String[] getStatsInfo(GamePlayer player) throws PlayerNotExistException;
024
025    /**
026     * 獲取遊戲玩家資訊
027     * @param player 玩家
028     * @return 遊戲玩家資訊
029     * @throws PlayerNotExistException 遊戲玩家不存在時
030     */
031    String[] getStatsInfo(Player player) throws PlayerNotExistException;
032
033    /**
034     * 增加殺數
035     * @param player 玩家
036     * @param kills 增加
037     */
038    void addKills(GamePlayer player, int kills);
039
040    /**
041     * 減少玩家殺數
042     * @param player 玩家
043     * @param kills 減少
044     */
045    void minusKills(GamePlayer player, int kills);
046
047    /**
048     * 增加玩家死數
049     * @param player 玩家
050     * @param deaths 增加
051     */
052    void addDeaths(GamePlayer player, int deaths);
053
054    /**
055     * 減少玩家死數
056     * @param player 玩家
057     * @param deaths 減少
058     */
059    void minusDeaths(GamePlayer player, int deaths);
060
061    /**
062     * 增加玩家勝數
063     * @param player 玩家
064     * @param wins 增加
065     */
066    void addWins(GamePlayer player, int wins);
067
068    /**
069     * 減少玩家勝數
070     * @param player 玩家
071     * @param wins 減少
072     */
073    void minusWins(GamePlayer player, int wins);
074
075    /**
076     * 增加玩家遊玩次數
077     * @param player 玩家
078     * @param played 增加
079     */
080    void addPlayed(GamePlayer player, int played);
081
082    /**
083     * 減少玩家遊玩次數
084     * @param player 玩家
085     * @param played 減少
086     */
087    void minusPlayed(GamePlayer player, int played);
088
089    /**
090     * 新增分數
091     * @param player 玩家
092     * @param scores 分數
093     */
094    void addScores(GamePlayer player, double scores);
095
096    /**
097     * 減少分數
098     * @param player 玩家
099     * @param scores 分數
100     */
101    void minusScores(GamePlayer player, double scores);
102
103    /**
104     * 獲取 遊戲玩家資訊的 可編輯容器
105     * @param player 玩家
106     * @return 遊戲玩家資訊
107     * @throws PlayerNotExistException 玩家不存在時
108     */
109    GameStatsEditor getGameStats(GamePlayer player) throws PlayerNotExistException;
110
111    /**
112     * 加載該玩家的遊戲資料
113     * @param player 玩家
114     * @return 異步運行
115     */
116    CompletableFuture<Boolean> loadGameStats(GamePlayer player);
117
118    /**
119     * 儲存所有玩家的遊戲資料
120     * @return 異步運行
121     */
122    CompletableFuture<Void> saveAll();
123
124    /**
125     * 保存單獨玩家的遊戲資料
126     * @param player 玩家
127     * @return 異步運行
128     * @throws PlayerNotExistException 玩家不存在時
129     */
130    CompletableFuture<Void> savePlayer(OfflinePlayer player) throws PlayerNotExistException;
131
132}
