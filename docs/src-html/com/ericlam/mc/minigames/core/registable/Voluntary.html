001package com.ericlam.mc.minigames.core.registable;
002
003import com.ericlam.mc.minigames.core.SectionTask;
004import com.ericlam.mc.minigames.core.game.GameTeam;
005import com.ericlam.mc.minigames.core.game.InGameState;
006import org.bukkit.inventory.ItemStack;
007
008/**
009 * 可選註冊器。不註冊并不會導致遊戲無法被激活。
010 */
011public interface Voluntary {
012
013    /**
014     * 註冊大堂倒數和遊戲結束之間的遊戲程序 (順序註冊)
015     *
016     * @param state 所屬場地狀態
017     * @param task  所屬遊戲程序
018     */
019    void registerGameTask(InGameState state, SectionTask task);
020
021    /**
022     * 添加等待投票物品
023     * @param slot slot
024     * @param item 物品
025     */
026    void addJoinItem(int slot, ItemStack item);
027
028    /**
029     * 添加遊戲內隊伍物品
030     * @param team 隊伍
031     * @param slot slot
032     * @param item 物品
033     */
034    void addGameItem(GameTeam team, int slot, ItemStack item);
035
036    /**
037     * 添加遊戲內非隊伍物品，僅限沒有隊伍玩家
038     * @param slot slot
039     * @param item 物品
040     */
041    void addGameItem(int slot, ItemStack item);
042
043    /**
044     * 添加觀戰隊伍物品
045     * @param team 隊伍
046     * @param slot slot
047     * @param item 物品
048     */
049    void addSpectatorItem(GameTeam team, int slot, ItemStack item);
050
051    /**
052     * 添加觀戰非隊伍物品, 僅限沒有隊伍玩家
053     * @param slot slot
054     * @param item 物品
055     */
056    void addSpectatorItem(int slot, ItemStack item);
057
058}
