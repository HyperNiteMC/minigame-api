001package com.ericlam.mc.minigames.core.arena;
002
003import com.google.common.collect.ImmutableMap;
004import org.bukkit.Location;
005import org.bukkit.World;
006
007import javax.annotation.Nonnull;
008import java.io.File;
009import java.util.concurrent.CompletableFuture;
010
011/**
012 * 用於被使用者實作來註冊場地機制運作
013 */
014public interface ArenaConfig {
015
016    /**
017     * @return 場地裝載資料夾
018     */
019    File getArenaFolder();
020
021    /**
022     *
023     * @return 最大場地加載數(等待中時)
024     */
025    int getMaxLoadArena();
026
027    /**
028     *
029     * @param world 世界設置
030     */
031    void setExtraWorldSetting(@Nonnull World world);
032
033    /**
034     * 獲得可設置的地標名稱和最大設置數
035     * <p>
036     * String 是地標名稱, Integer 是該地標最大設置數
037     * @return 允許的地標
038     */
039    ImmutableMap<String, Integer> getAllowWarps();
040
041    /**
042     *
043     * @return 等候房間位置
044     */
045    Location getLobbyLocation();
046
047    /**
048     * 當遊戲結束後自動回傳的伺服器
049     *
050     * @return 返回伺服器的名稱
051     */
052    String getFallBackServer();
053
054    /**
055     *
056     * @return 此小遊戲的前綴
057     */
058    String getGamePrefix();
059
060    /**
061     * 如何儲存等候房間位置到你的 yaml
062     *
063     * @param location 等候房間位置
064     * @return 設置是否成功
065     */
066    CompletableFuture<Boolean> setLobbyLocation(final Location location);
067
068}
