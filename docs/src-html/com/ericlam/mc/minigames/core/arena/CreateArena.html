001package com.ericlam.mc.minigames.core.arena;
002
003import com.ericlam.mc.minigames.core.exception.NoMoreElementException;
004import com.ericlam.mc.minigames.core.exception.arena.create.LocationMaxReachedException;
005import com.ericlam.mc.minigames.core.exception.arena.create.NoMoreLocationException;
006import com.ericlam.mc.minigames.core.exception.arena.create.WarpExistException;
007import com.ericlam.mc.minigames.core.exception.arena.create.WarpNotExistException;
008import org.bukkit.Location;
009import org.bukkit.World;
010
011import java.util.LinkedList;
012import java.util.List;
013import java.util.Map;
014import java.util.Optional;
015
016/**
017 * 設置場地接口, 用於設置場地
018 */
019public interface CreateArena extends Arena {
020
021    /**
022     * @param author 場地作者
023     */
024    void setAuthor(String author);
025
026    /**
027     *
028     * @param world 場地世界
029     */
030    void setWorld(World world);
031
032    /**
033     *
034     * @param arenaName 場地名稱
035     */
036    void setArenaName(String arenaName);
037
038    /**
039     *
040     * @param displayName 場地顯示名稱
041     */
042    void setDisplayName(String displayName);
043
044    /**
045     *
046     * @param warp 地標
047     * @param location 位置
048     * @param max 最大設置數
049     * @throws WarpNotExistException 地標不存在時
050     * @throws LocationMaxReachedException 地標位置數過多時
051     */
052    default void addLocation(String warp, Location location, final int max) throws WarpNotExistException, LocationMaxReachedException {
053        List<Location> locs = Optional.ofNullable(getWarp(warp)).orElseThrow(() -> new WarpNotExistException(warp));
054        if (locs.size() >= max) throw new LocationMaxReachedException(warp);
055        locs.add(location);
056    }
057
058    /**
059     * 刪除上一個地標
060     * @param warp 地標
061     * @throws WarpNotExistException 地標不存在時
062     * @throws NoMoreLocationException 沒有更多位置時
063     */
064    default void removeLastLocation(String warp) throws WarpNotExistException, NoMoreLocationException {
065        List<Location> locs = Optional.ofNullable(getWarp(warp)).orElseThrow(() -> new WarpNotExistException(warp));
066        if (locs.size() == 0) throw new NoMoreLocationException(warp);
067        locs.remove(locs.size() - 1);
068    }
069
070    /**
071     *
072     * @param warp 添加地標
073     * @throws WarpExistException 地標已存在時
074     */
075    default void addWarp(String warp) throws WarpExistException {
076        if (getLocationsMap().putIfAbsent(warp, new LinkedList<>()) != null) throw new WarpExistException(warp);
077    }
078
079    /**
080     *
081     * @param warp 刪除地標
082     * @throws WarpNotExistException 地標不存在時
083     */
084    default void removeWarp(String warp) throws WarpNotExistException {
085        if (getLocationsMap().remove(warp) == null) throw new WarpNotExistException(warp);
086    }
087
088    /**
089     * 為描述添加新一行
090     * @param text 描述文字
091     */
092    default void addDescriptionLine(String text) {
093        getDescription().add(text);
094    }
095
096    /**
097     * 刪除上一行描述
098     * @throws NoMoreElementException 沒有更多描述時
099     */
100    default void removeDescriptionLine() throws NoMoreElementException {
101        if (getDescription().size() == 0) throw new NoMoreElementException(this.getArenaName());
102        getDescription().remove(getDescription().size() - 1);
103    }
104
105    /**
106     *
107     * @param locationMap 地標列表
108     */
109    void setLocationMap(Map<String, List<Location>> locationMap);
110
111    /**
112     *
113     * @return 場地有否被變更
114     */
115    boolean isChanged();
116
117    /**
118     * 設置場地有否被變更
119     * @param changed 有否被變更
120     */
121    void setChanged(Boolean changed);
122
123    /**
124     * 定義場地完成設置的條件
125     * @return 是否完成設置
126     */
127    boolean isSetupCompleted();
128}
