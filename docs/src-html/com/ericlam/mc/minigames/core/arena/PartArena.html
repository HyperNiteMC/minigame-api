001package com.ericlam.mc.minigames.core.arena;
002
003import org.bukkit.util.BlockVector;
004
005/**
006 * 需要標註區域範圍的場地專用接口
007 * <p>
008 * 頂端位置與低端位置，將會形成一個長方體的場地區域。
009 */
010public interface PartArena extends Arena {
011
012    /**
013     * @return 場地區域最頂端的位置
014     */
015    BlockVector getTopCorner();
016
017    /**
018     *
019     * @return 場地區域最低端的位置
020     */
021    BlockVector getBottomCorner();
022
023}
