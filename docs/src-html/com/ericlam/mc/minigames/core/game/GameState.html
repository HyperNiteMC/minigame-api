001package com.ericlam.mc.minigames.core.game;
002
003import net.md_5.bungee.api.ChatColor;
004
005/**
006 * 遊戲狀態
007 */
008public enum GameState {
009
010    /**
011     * 投票中
012     */
013    VOTING("&a等待中"),
014
015    /**
016     * 即將開始
017     */
018    PRESTART("&b遊戲即將開始"),
019
020    /**
021     * 遊戲中
022     */
023    IN_GAME("&b遊戲中"),
024
025    /**
026     * 邁向完結
027     */
028    PREEND("&c遊戲邁向完結"),
029
030    /**
031     * 遊戲完結; 設置這個狀態後將會直接重啟伺服器
032     */
033    ENDED("&c遊戲已完結"),
034
035    /**
036     * 遊戲沒有被激活
037     */
038    STOPPED(null);
039
040    private String motd;
041
042    GameState(final String motd) {
043        this.motd = motd;
044    }
045
046    /**
047     * 獲取 motd
048     *
049     * @return motd
050     */
051    public String getMotd() {
052        return ChatColor.translateAlternateColorCodes('&', motd);
053    }
054
055}
