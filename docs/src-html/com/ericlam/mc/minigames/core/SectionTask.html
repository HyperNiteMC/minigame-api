001package com.ericlam.mc.minigames.core;
002
003import com.ericlam.mc.minigames.core.manager.PlayerManager;
004
005/**
006 * 遊戲運行程序接口
007 */
008public interface SectionTask {
009
010    /**
011     * 定義初始化時的動作
012     *
013     * @param playerManager 玩家管理器
014     */
015    void initTimer(PlayerManager playerManager);
016
017    /**
018     * 在取消程序時運行
019     */
020    void onCancel();
021
022    /**
023     * 在完成程序時運行
024     */
025    void onFinish();
026
027    /**
028     * 每秒的運行程序
029     * @param remain 剩餘秒數
030     * @return 最終的剩餘秒數
031     */
032    long run(long remain);
033
034    /**
035     * 獲取這個運行程序的總秒數
036     * @return 總秒數
037     */
038    long getTotalTime();
039
040    /**
041     * 取消此運行程序的條件
042     * @return 取消此運行程序的條件
043     */
044    boolean shouldCancel();
045
046    /**
047     * 判斷程序是否在運行
048     * @return 是否在運行
049     */
050    boolean isRunning();
051
052    /**
053     * 設置程序是否在運行
054     * @param running 是否在運行
055     */
056    void setRunning(boolean running);
057}
